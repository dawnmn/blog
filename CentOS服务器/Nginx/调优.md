#### **内核调优**
```
vim /etc/sysctl.conf

fs.file-max = 999999	#一个进程可以打开的最大句柄数
net.ipv4.tcp_tw_reuse = 1	#重用TIME_WAIT状态的TCP连接
net.ipv4.tcp_keepalive_time = 600	#TCP保持连接的最大时间
net.ipv4.tcp_fin_timeout = 30	#TCP 四次挥手的FIN_WAIT2状态的最大时间
net.ipv4.tcp_max_tw_buckets = 5000	#保持在TIME_WAIT状态套接字的最大数目
net.ipv4.ip_local_port_range = 1024 61000 #本地端口取值范围
net.ipv4.tcp_rmem = 4096 32768 262142	#TCP接收缓存的最小值 默认值 最大值
net.ipv4.tcp_wmem = 4096 32768 262142	#TCP发送缓存的最小值 默认值 最大值
net.core.netdev_max_backlog = 8096	#
net.core.rmem_default = 262144
net.core.wmem_default = 262144
net.core.rmem_max = 2097152
net.core.wmem_max = 2097152
net.ipv4.tcp_syncookies = 1	#解决TCP的syn攻击
net.ipv4.tcp_max_syn.backlog=1024	#接收SYN请求队列的最大长度
```
