class SortUnit
{
    public static function bubble(array &$arr){
        $count = count($arr);
        for($i=0;$i<$count-1;$i++){
            $exchange = false;
            for($j=0;$j<$count-$i-1;$j++){
                if($arr[$j]>$arr[$j+1]){
                    $exchange = true;
                    $temp = $arr[$j];
                    $arr[$j] = $arr[$j+1];
                    $arr[$j+1] = $temp;
                }
            }
            if(!$exchange){
                return;
            }
        }
    }
 
    public static function select(array &$arr){
        $count = count($arr);
        for($i=0;$i<$count-1;$i++){
            $min = $i;
            for($j=$i+1;$j<$count;$j++){
                if($arr[$min] > $arr[$j]){
                    $min = $j;
                }
            }
            $temp = $arr[$i];
            $arr[$i] = $arr[$min];
            $arr[$min] = $temp;
        }
    }

	public static function insert(array &$arr){
        $count = count($arr);
        for($i=0;$i<$count-1;$i++){
            for($j=$i+1;$j>0;$j--){
                if($arr[$j] < $arr[$j-1]){
                    $temp = $arr[$j-1];
                    $arr[$j-1] = $arr[$j];
                    $arr[$j] = $temp;
                }
            }
        }
    }
 
    public static function quick(array &$arr, int $left, int $right){
        if($left >= $right){
            return;
        }
        $pivot = $arr[$left];
        $i = $left;
        $j = $right+1;
        while (true){
            while ($i < --$j) if($pivot > $arr[$j]) break;
            while (++$i < $j) if($pivot < $arr[$i]) break;
            if($i >= $j) break;
            $temp = $arr[$i];
            $arr[$i] = $arr[$j];
            $arr[$j] = $temp;
        }
        $arr[$left] = $arr[$j];
        $arr[$j] = $pivot;
        self::quick($arr, $left, $j-1);
        self::quick($arr, $j+1, $right);
    }
 
    public static function shell(&$arr)
    {
        for ($gap = count($arr) >> 1; $gap > 0; $gap >>= 1)
            for ($i = $gap; $i < count($arr); $i++) {
                $temp = $arr[$i];
                for ($j = $i - $gap; $j >= 0 && $arr[$j] > $temp; $j -= $gap)
                    $arr[$j + $gap] = $arr[$j];
                $arr[$j + $gap] = $temp;
            }
    }
}
