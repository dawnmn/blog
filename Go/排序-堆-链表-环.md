**排序 sort**
实现了4种基本排序算法：插入排序、归并排序、堆排序和快速排序，sort会根据输入数据自动选择。

```
func Ints(x []int) { Sort(IntSlice(x)) }
func Float64s(x []float64) { Sort(Float64Slice(x)) }
func Strings(x []string) { Sort(StringSlice(x)) }

type Interface interface {
	// 获取数据集合元素个数
	Len() int
	// 如果 i 索引的数据小于 j 索引的数据，返回 true，且不会调用下面的 Swap()，即数据升序排序。
	Less(i, j int) bool
	// 交换 i 和 j 索引的两个元素的位置
	Swap(i, j int)
}
```
`sort.Search` 仅用于有序列表，在有序列表中使用二分查找满足条件的最小索引
```
a := []int{1, 2, 3, 4, 5}
d := sort.Search(len(a), func(i int) bool { return a[i] == 3 })
fmt.Println(d) // 2
```

int切片逆序排序示例：
```
num := []int{70, 80, 20, 50, 10}
sort.Sort(sort.Reverse(sort.IntSlice(num)))
```

结构体排序示例：
// 学生成绩结构体
```
type StuScore struct {
	name  string // 姓名
	score int    // 成绩
}

type StuScores []StuScore

//Len()
func (s StuScores) Len() int {
	return len(s)
}

//Less(): 成绩将有低到高排序
func (s StuScores) Less(i, j int) bool {
	return s[i].score < s[j].score
}

//Swap()
func (s StuScores) Swap(i, j int) {
	s[i], s[j] = s[j], s[i]
}

func main() {
	stus := StuScores{{"alan", 95}, {"hikerell", 91}, {"acmfly", 96}, {"leao", 90}}
	sort.Sort(stus)
	// 逆序排序
	sort.Sort(sort.Reverse(stus))
}
```
**堆**
```
type IntHeap []int

func (h IntHeap) Len() int           { return len(h) }
func (h IntHeap) Less(i, j int) bool { return h[i] < h[j] }
func (h IntHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }

func (h *IntHeap) Push(x interface{}) {
   *h = append(*h, x.(int))
}

func (h *IntHeap) Pop() interface{} {
   old := *h
   n := len(old)
   x := old[n-1]
   *h = old[0 : n-1]
   return x
}
```

**双向链表**
```
type Element struct {
    next, prev *Element  // 上一个元素和下一个元素
    list *List  // 元素所在链表
    Value interface{}  // 元素
}

type List struct {
    root Element  // 链表的根元素
    len  int      // 链表的长度
}
```

示例：
```
func main() {
	list := list.New()
	// 在链表最后插入元素
	list.PushBack(1)
	list.PushBack(2)

	fmt.Printf("len: %v\n", list.Len())
	// 第一个元素
	fmt.Printf("first: %#v\n", list.Front())
	fmt.Printf("second: %#v\n", list.Front().Next())
}
```

**环**
```
type Ring struct {
    next, prev *Ring
    Value      interface{}
}
```

示例：
```
func main() {
	ring := ring.New(3)

	for i := 1; i <= 3; i++ {
		ring.Value = i
		ring = ring.Next()
	}

	// 计算 1+2+3
	s := 0
	ring.Do(func(p interface{}) {
		s += p.(int)
	})
}
```
